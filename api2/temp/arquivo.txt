async fn download(req: HttpRequest) -> Result<HttpResponse> {
    // Define o caminho para o arquivo que será baixado
    let file_path: PathBuf = "./path/to/your/file.txt".into();

    // Abra o arquivo
    let mut file = File::open(&file_path).await.map_err(actix_web::error::ErrorInternalServerError)?;

    // Lê o conteúdo do arquivo para um buffer
    let mut buffer = Vec::new();
    file.read_to_end(&mut buffer).await.map_err(actix_web::error::ErrorInternalServerError)?;

    // Cria a resposta com o conteúdo do arquivo
    Ok(HttpResponse::Ok()
        .content_type("application/octet-stream")
        .insert_header(("Content-Disposition", "attachment; filename=file.txt"))
        .body(buffer))
}
